        // Game Logic
        const canvas = document.getElementById('gameCanvas');
        const ctx = canvas.getContext('2d');
        
        let gameRunning = false;
        let score = 0;
        let highScore = 0;
        let animationId;

        // Responsive canvas
        function resizeCanvas() {
            const maxWidth = Math.min(800, window.innerWidth - 40);
            canvas.width = maxWidth;
            canvas.height = maxWidth * 0.5;
        }

        // Player
        const player = {
            x: 50,
            y: canvas.height - 60,
            width: 30,
            height: 30,
            velocityY: 0,
            jumping: false,
            color: '#60a5fa'
        };

        // Obstacles
        let obstacles = [];
        let obstacleSpeed = 5;
        let obstacleInterval = 90;
        let frameCount = 0;

        // Ground
        const groundY = canvas.height - 30;

        function drawPlayer() {
            ctx.fillStyle = player.color;
            ctx.shadowBlur = 15;
            ctx.shadowColor = player.color;
            ctx.fillRect(player.x, player.y, player.width, player.height);
            ctx.shadowBlur = 0;
        }

        function drawObstacle(obs) {
            ctx.fillStyle = '#ef4444';
            ctx.shadowBlur = 15;
            ctx.shadowColor = '#ef4444';
            ctx.fillRect(obs.x, obs.y, obs.width, obs.height);
            ctx.shadowBlur = 0;
        }

        function drawGround() {
            ctx.fillStyle = 'rgba(96, 165, 250, 0.2)';
            ctx.fillRect(0, groundY, canvas.width, 30);
        }

        function drawScore() {
            ctx.fillStyle = '#93c5fd';
            ctx.font = 'bold 24px Arial';
            ctx.fillText('Score: ' + score, 20, 40);
        }

        function jump() {
            if (!player.jumping && player.y >= groundY - player.height) {
                player.velocityY = -15;
                player.jumping = true;
            }
        }

        function updatePlayer() {
            player.velocityY += 0.8; // Gravity
            player.y += player.velocityY;

            if (player.y >= groundY - player.height) {
                player.y = groundY - player.height;
                player.velocityY = 0;
                player.jumping = false;
            }
        }

        function createObstacle() {
            const height = Math.random() * 40 + 30;
            obstacles.push({
                x: canvas.width,
                y: groundY - height,
                width: 25,
                height: height
            });
        }

        function updateObstacles() {
            frameCount++;
            
            if (frameCount % obstacleInterval === 0) {
                createObstacle();
            }

            for (let i = obstacles.length - 1; i >= 0; i--) {
                obstacles[i].x -= obstacleSpeed;

                if (obstacles[i].x + obstacles[i].width < 0) {
                    obstacles.splice(i, 1);
                    score++;
                    document.getElementById('scoreDisplay').textContent = score;
                    
                    // Increase difficulty
                    if (score % 10 === 0) {
                        obstacleSpeed += 0.5;
                        obstacleInterval = Math.max(60, obstacleInterval - 5);
                    }
                }
            }
        }

        function checkCollision() {
            for (let obs of obstacles) {
                if (player.x < obs.x + obs.width &&
                    player.x + player.width > obs.x &&
                    player.y < obs.y + obs.height &&
                    player.y + player.height > obs.y) {
                    return true;
                }
            }
            return false;
        }

        function gameLoop() {
            if (!gameRunning) return;

            ctx.clearRect(0, 0, canvas.width, canvas.height);
            
            drawGround();
            drawPlayer();
            drawScore();
            
            obstacles.forEach(drawObstacle);
            
            updatePlayer();
            updateObstacles();

            if (checkCollision()) {
                gameOver();
                return;
            }

            animationId = requestAnimationFrame(gameLoop);
        }

        function startGame() {
            if (gameRunning) return;
            
            gameRunning = true;
            score = 0;
            obstacles = [];
            obstacleSpeed = 5;
            obstacleInterval = 90;
            frameCount = 0;
            player.y = groundY - player.height;
            player.velocityY = 0;
            player.jumping = false;
            
            document.getElementById('scoreDisplay').textContent = score;
            document.getElementById('startBtn').textContent = 'Playing...';
            
            gameLoop();
        }

        function gameOver() {
            gameRunning = false;
            cancelAnimationFrame(animationId);
            
            if (score > highScore) {
                highScore = score;
                document.getElementById('highScoreDisplay').textContent = highScore;
            }
            
            document.getElementById('startBtn').textContent = 'Play Again';
            
            ctx.fillStyle = 'rgba(15, 23, 42, 0.8)';
            ctx.fillRect(0, 0, canvas.width, canvas.height);
            ctx.fillStyle = '#ef4444';
            ctx.font = 'bold 48px Arial';
            ctx.textAlign = 'center';
            ctx.fillText('GAME OVER', canvas.width / 2, canvas.height / 2 - 20);
            ctx.fillStyle = '#60a5fa';
            ctx.font = 'bold 24px Arial';
            ctx.fillText('Score: ' + score, canvas.width / 2, canvas.height / 2 + 30);
            ctx.textAlign = 'left';
        }

        function resetGame() {
            gameRunning = false;
            cancelAnimationFrame(animationId);
            score = 0;
            obstacles = [];
            obstacleSpeed = 5;
            obstacleInterval = 90;
            frameCount = 0;
            player.y = groundY - player.height;
            player.velocityY = 0;
            player.jumping = false;
            
            document.getElementById('scoreDisplay').textContent = score;
            document.getElementById('startBtn').textContent = 'Start Game';
            
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            drawGround();
            drawPlayer();
        }

        // Controls
        document.addEventListener('keydown', (e) => {
            if ((e.code === 'Space' || e.code === 'ArrowUp') && gameRunning) {
                e.preventDefault();
                jump();
            }
        });

        canvas.addEventListener('click', () => {
            if (gameRunning) jump();
        });

        canvas.addEventListener('touchstart', (e) => {
            e.preventDefault();
            if (gameRunning) jump();
        });

        // Initial draw
        resetGame();
    </script>
